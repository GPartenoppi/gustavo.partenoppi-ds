import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.println("Digite dois números:");
        int a = sc.nextInt();
        int b = sc.nextInt();

        if(a > 0 && b > 0)
            System.out.println("Soma: " + (a + b));
        else
            System.out.println("Valores inválidos");
    }
}
